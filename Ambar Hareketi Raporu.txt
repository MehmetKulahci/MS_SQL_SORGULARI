SELECT 
 CASE STF.TRCODE WHEN 25 THEN 'Ambar Fişi'
WHEN 11 THEN 'Fire Fişi'
WHEN 13 THEN 'Üretimden Giriş Fişi'
WHEN 12 THEN 'Ambar Fişi'
END [FIS_TURU],

CONVERT(TIME,STF.CAPIBLOCK_CREADEDDATE) [FIS_OLUSTURMA_SAATI],

LBL.SERIALNO [BARKOD],
V.NAME [VARYANT],
STL.AMOUNT*STL.UINFO6 [M2],
STL.AMOUNT [PK],

DBO.TMP_PEKETICI(STL.VARIANTREF) [PAKET İÇİ],
dbo.OzellikDegeri_013(V.ITEMREF,V.LOGICALREF,'QUALITY',1) [KALİTE],
U.NAME [KULLANICI_ADI],
C.NAME [AMBAR]

FROM LG_013_01_STLINE STL
LEFT JOIN LG_013_01_STFICHE STF WITH (NOLOCK) ON STL.STFICHEREF=STF.LOGICALREF
INNER JOIN PRODUZCE..STLINE S  WITH (NOLOCK) ON S.STFICHEREF=STF.LOGICALREF
INNER JOIN PRODUZCE..LABEL_CARD LBL WITH (NOLOCK) ON LBL.LOGICALREF=S.BARCODEREF
INNER JOIN LG_013_ITEMS ITM  WITH (NOLOCK) ON ITM.LOGICALREF=STL.STOCKREF
INNER JOIN LG_013_VARIANT V WITH (NOLOCK) ON STL.VARIANTREF=V.LOGICALREF
INNER JOIN L_CAPIWHOUSE C WITH (NOLOCK) ON C.NR=STL.SOURCEINDEX
INNER JOIN L_CAPIUSER U WITH (NOLOCK) ON U.LOGICALREF=STF.CAPIBLOCK_CREATEDBY

WHERE STF.TRCODE IN (25,11,12,13)
